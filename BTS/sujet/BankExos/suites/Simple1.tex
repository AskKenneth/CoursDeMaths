\exo{Soit la suite $(u_n)$ de terme général $u_n = \dfrac{1}{n+1}$}

\question{}
Exprimer $u_{n+1} - u_n$ en fonction de $n$.

\correction{
$u_{n+1} - u_n = \dfrac{1}{n+2} - \dfrac{1}{n+1} = \dfrac{(n+1)-(n+2)}{(n+1)(n+2)} = -\dfrac{1}{(n+1)(n+2)}$

Or, $n \geq 0$ donc $n+1 > 0$ et $n+2 > 0$, donc $u_{n+1} - u_n <0$
}

\question{}
En déduire le sens de variation de la suite $(u_n)$.

\correction{
$u_{n+1} - u_n < 0 \Leftrightarrow u_{n+1} < u_n$

La suite $(u_n)$ est donc décroissante.
}
